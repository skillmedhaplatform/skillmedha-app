"use client";
import {
  Button,
  Checkbox,
  Col,
  message,
  Modal,
  Row,
  Select,
  Space,
  Switch,
  Table,
  Upload,
  Input,
} from "antd";
import Column from "antd/es/table/Column";
import React, { useEffect, useState } from "react";
import { useSelector, useDispatch } from "react-redux";
import userCompStyles from "../styles/usercomp.module.scss";
import { Drawer } from "antd";
import { saveAs } from "file-saver";
import * as XLSX from "xlsx";
import Papa from "papaparse";
import {
  BlockStudent,
  clearSearchStudent,
  createStudent,
  getStudent,
  searchStudent,
  UnBlockStudent,
} from "@/app/redux/slice/students";
import { getTests, SendEmailMail, updateTest } from "@/app/redux/slice/test";
import { v4 as uuidv4 } from "uuid";
import { kycUrl, skillmedhaTestPortal } from "@/app/utils/urls";
import { encryptObject } from "@/app/middlewares/encryptedData";
import { generateKycEmailTemplate } from "../emailtemplate/kycLinkEmailui";
import { generatePasswordEmailTemplate } from "../emailtemplate/register";
import { getLstorage } from "@/app/middlewares/windowMW";

const { Search } = Input;

const UsersComp = () => {
  const allTests = useSelector((state) => state.tests.value);
  const getAllTestStatus = useSelector(
    (state) => state.tests.getAllTestStatus.status
  );
  const allStudents = useSelector((state) => state.students.allStudents);
  const allStudentsStatus = useSelector((state) => state.students.status);
  const searchedStudent = useSelector(
    (state) => state.students.serchedStudent.value
  );

  // Pagination state: current page + page size (with size changer)
  const [pagination, setPagination] = useState({ current: 1, pageSize: 10 });

  const Students = searchedStudent.length ? searchedStudent : allStudents;

  const student = useSelector((state) => state.students.student);
  const [selectedRowKeys, setSelectedRowKeys] = useState([]);
  const [searchText, setSearchText] = useState("");
  const [drawer, setDrawer] = useState(false);
  const [addUser, setAddUser] = useState(false);
  const dispatch = useDispatch();
  const [studentCreds, setStudentCreds] = useState({});
  const [modalBlock, setModalBlock] = useState(false);
  const [selectedTestId, setSelectedTestId] = useState(null);
  const [selectedStudentId, setSelectedStudentId] = useState(null);
  const [faceDataFilter, setFaceDataFilter] = useState("all"); // ✅ NEW STATE

  useEffect(() => {
    if (!allTests?.length) dispatch(getTests({ cursor: null, limit: 100 }));
  }, []);

  const handleCaptureFaceData = (record) => {
    const key = encryptObject({
      object: { createdAt: record.createdAt },
      key: new Date(record.createdAt).toString(),
    });
    const token = encryptObject({
      object: {
        email: record.email,
        studentId: record?.globalId,
      },
      key,
    });

    const url =
      skillmedhaTestPortal +
      `?s_tk=${token}&s_tks=${Math.random()
        .toString(32)
        .slice(3)}sGIR9HD${Math.random()
        .toString(32)
        .slice(3)}K+R3tzKcQm0sfT/emKZ4YdA==&s_t=${key}&st=${getLstorage(
        "token"
      )}`;
    const emailData = {
      from: "cloud.admin@ksquareinfo.com",
      to: record.email,
      subject: "KYC Link",
      text: `Your KYC link is: ${url}`,
      html: generateKycEmailTemplate({
        userName: record?.email,
        url: url,
        companyDetails: "Skill Medha",
      }),
    };
    message.success("KYC Link sent");
    dispatch(SendEmailMail({ updates: emailData }));
  };

  // ✅ UPDATED FILTERING LOGIC
  const filteredStudents = Students.filter((student) => {
    const email = student?.email?.toLowerCase() || "";
    const username = student?.userName?.toLowerCase() || "";
    const search = searchText.toLowerCase();
    const matchesSearch = email.includes(search) || username.includes(search);

    let matchesFaceData = true;
    if (faceDataFilter === "complete") {
      matchesFaceData = !!student?.faceData;
    } else if (faceDataFilter === "incomplete") {
      matchesFaceData = !student?.faceData;
    }

    return matchesSearch && matchesFaceData;
  });

  // Keep current and pageSize in sync with Table pagination UI
  const handleTableChange = (newPagination) => {
    setPagination((prev) => ({
      ...prev,
      current: newPagination.current,
      pageSize: newPagination.pageSize,
    }));
  };

  return (
    <div className={userCompStyles.mainContainer}>
      <div className={userCompStyles.head}>
        <div className={userCompStyles.headLeft}>
          <Search
            placeholder="Search by Email or Username"
            allowClear
            value={searchText}
            onChange={(e) => setSearchText(e.target.value)}
            style={{ width: 300 }}
          />
        </div>
        <div className={userCompStyles.headRight}>
          <Select
            style={{ width: "15rem" }}
            value={faceDataFilter}
            onChange={(value) => setFaceDataFilter(value)}
          >
            <Select.Option value="all">All Students</Select.Option>
            <Select.Option value="complete">Face Data Complete</Select.Option>
            <Select.Option value="incomplete">
              Face Data Incomplete
            </Select.Option>
          </Select>
        </div>
      </div>

      <div className={userCompStyles.container}>
        <Table
          loading={allStudentsStatus == "pending"}
          dataSource={filteredStudents}
          pagination={{
            current: pagination.current,
            pageSize: pagination.pageSize,
            total: filteredStudents.length,
            showSizeChanger: true,
            pageSizeOptions: [10, 20, 30, 40, 50],
            showTotal: (total, range) => `${range}-${range[1]} of ${total}`,
          }}
          onChange={handleTableChange}
          rowKey="_id"
          locale={{ emptyText: "No students found" }}
        >
          <Column
            title="S.No"
            key="serial"
            render={(_, __, index) =>
              (pagination.current - 1) * pagination.pageSize + index + 1
            }
          />
          <Column
            title="Name"
            key="name"
            render={(_, record) => {
              const fullName =
                (record?.firstName || "") + " " + (record?.lastName || "");

              return <div>{fullName.trim() || "N/A"}</div>;
            }}
          />
          <Column
            title="Email"
            key="email"
            render={(_, record) => <div>{record?.email || "N/A"}</div>}
          />
          <Column
            title="Phone"
            key="phone"
            render={(_, record) => <div>{record?.phone || "N/A"}</div>}
          />
          <Column
            title="User Name"
            key="username"
            render={(_, record) => <div>{record?.userName || "N/A"}</div>}
          />
          <Column
            title="Face Data"
            key="facedata"
            render={(_, record) => (
              <div>
                {record?.faceData ? (
                  <div
                    style={{
                      display: "flex",
                      flexDirection: "column",
                      alignItems: "flex-start",
                      justifyContent: "flex-start",
                      gap: ".5rem",
                    }}
                  >
                    <img
                      src={record?.faceData?.file}
                      className={userCompStyles.faceData}
                    />
                    <Button
                      style={{ width: "70%" }}
                      type="primary"
                      onClick={() => handleCaptureFaceData(record)}
                    >
                      Resend KYC Link
                    </Button>
                  </div>
                ) : (
                  <Button
                    type="primary"
                    onClick={() => handleCaptureFaceData(record)}
                  >
                    Send KYC Link
                  </Button>
                )}
              </div>
            )}
          />
        </Table>
      </div>

      <Modal
        open={modalBlock}
        onCancel={() => setModalBlock(false)}
        onOk={() => {
          dispatch(
            UnBlockStudent({
              testId: selectedTestId,
              studentId: selectedStudentId,
              dispatch,
            })
          );
          setModalBlock(false);
        }}
        title={<h5>The Student is blocked for this Test</h5>}
        okText={"Yes"}
        cancelText="No"
      >
        <p>Are you sure you want to Unblock the student for this test?</p>
      </Modal>
    </div>
  );
};

export default UsersComp;
