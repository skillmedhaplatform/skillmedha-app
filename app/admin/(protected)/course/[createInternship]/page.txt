"use client";
import React, { useEffect, useState } from "react";
import internStyles from "./page.module.scss";
import TextEditor from "@/utils/editor";
import ChipInput from "@/utils/chipInput/chip";
import { Button, Radio, Upload, Modal, message } from "antd";
import { PlusOutlined } from "@ant-design/icons";
import dynamic from "next/dynamic";
import { useParams, useRouter } from "next/navigation";
import { useDispatch, useSelector } from "react-redux";
import {
  createCourse,
  getOneInternship,
  updateInternship,
} from "@/redux/slices/cms/internship";
import { uploadFileToS3 } from "@/utils/uploadtos3";
import { restUrl } from "@/config/urls";

// IMPORTANT: client-only import so the crop modal can mount properly in Next.js
const ImgCrop = dynamic(() => import("antd-img-crop"), { ssr: false });

const Page = () => {
  const nav = useRouter();
  const params = useParams();
  const dispatch = useDispatch();

  const [mounted, setMounted] = useState(false);

  // existing fields
  const [title, setTitle] = useState("");
  const [description, setDescription] = useState("");
  const [preRequisites, setPreRequisites] = useState([]);
  const [difficulty, setDifficulty] = useState("Beginner");
  const [price, setPrice] = useState("");

  // image state
  const [coverImage, setCoverImage] = useState("");
  const [fileList, setFileList] = useState([]);
  const [previewOpen, setPreviewOpen] = useState(false);
  const [previewImage, setPreviewImage] = useState("");

  const userCreds = useSelector((state) => state.user.singleUser);

  useEffect(() => {
    if (params.createInternship?.split("/")?.join("") !== "newCourse") {
      dispatch(
        getOneInternship({
          id: params.createInternship,
          orgId: userCreds?.orgId,
        })
      )?.then((resp) => {
        if (resp?.payload) {
          setTitle(resp?.payload?.title || "");
          setDescription(resp?.payload?.description || "");
          setPreRequisites(resp?.payload?.preRequisites || []);
          setDifficulty(resp?.payload?.difficulty || "Beginner");
          setPrice(resp?.payload?.price || "");
          if (resp?.payload?.coverImage) {
            setCoverImage(resp.payload.coverImage);
            setFileList([
              {
                uid: "-1",
                name: "cover-image",
                status: "done",
                url: resp.payload.coverImage,
              },
            ]);
          }
        }
      });
    }
  }, [params.createInternship, dispatch, userCreds?.orgId]);

  useEffect(() => {
    setMounted(true);
  }, []);

  if (!mounted) return null;

  const validateBeforeCrop = (file) => {
    const isAllowed =
      file.type === "image/jpeg" ||
      file.type === "image/png" ||
      file.type === "image/webp";
    if (!isAllowed) {
      message.error("Only JPG/PNG/WebP images are allowed");
      return false;
    }
    const isLt2M = file.size / 1024 / 1024 < 2;
    // if (!isLt2M) {
    //   message.error("Image must be smaller than 2MB");
    //   return false;
    // }
    return true;
  };

  const getBase64 = (file) =>
    new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.readAsDataURL(file);
      reader.onload = () => resolve(reader.result);
      reader.onerror = (error) => reject(error);
    });

  const handlePreview = async (file) => {
    if (!file.url && !file.preview) {
      file.preview = await getBase64(file.originFileObj);
    }
    setPreviewImage(file.url || file.preview);
    setPreviewOpen(true);
  };

  // Confirm cropper opens first with this in place,
  // then keep customRequest to upload the CROPPED blob to S3.
  const handleCustomUpload = async (options) => {
    const { file, onSuccess, onError } = options;
    try {
      const url = await uploadFileToS3({
        file, // this is the cropped file from ImgCrop
        restUrl,
        onUploaded: (s3Url) => {
          setCoverImage(s3Url);
        },
      });
      setFileList([
        {
          uid: file.uid || String(Date.now()),
          name: file.name,
          status: "done",
          url,
        },
      ]);
      onSuccess && onSuccess({ url }, file);
    } catch (err) {
      onError && onError(err);
    }
  };

  const handleSave = async () => {
    if (params.createInternship?.split("/")?.join("") !== "newCourse") {
      dispatch(
        updateInternship({
          id: params.createInternship,
          dispatch,
          data: {
            title,
            description,
            preRequisites,
            difficulty,
            price,
            coverImage,
          },
        })
      );
    } else {
      dispatch(
        createCourse({
          title,
          description,
          preRequisites,
          difficulty,
          price,
          coverImage,
        })
      )?.then((resp) => {
        const id = resp.payload.data.insertedId;
        nav.replace("/course/" + id);
      });
    }
  };

  return (
    <div className={internStyles.container}>
      <div className={internStyles.titleContainer}>
        <div className={internStyles.title}>Course Details</div>
      </div>

      <div className={internStyles.body}>
        <div className={internStyles.internTtle}>
          <span>Course Title *</span>
          <input
            placeholder="Course Title"
            value={title}
            onChange={(e) => setTitle(e.target.value)}
          />
        </div>

        <div className={internStyles.desc}>
          <div className={internStyles.sd}>Short Description *</div>
          <TextEditor
            name="description"
            initialContent={{ description }}
            editorFun={(val) => setDescription(val)}
          />
          <Button
            type="primary"
            style={{ marginTop: "1rem", alignSelf: "flex-end" }}
          >
            Generate using AI
          </Button>
        </div>

        {/* Cover Image with client-only ImgCrop */}
        <div className={internStyles.coverImage_container}>
          <div className={internStyles.sd}>Cover Image :</div>

          <ImgCrop
            aspect={16 / 9}
            rotate
            zoom
            quality={0.8}
            beforeCrop={validateBeforeCrop}
            cropperProps={{ restrictPosition: false }}
            modalTitle="Edit image"
          >
            <Upload
              listType="picture-card"
              fileList={fileList}
              maxCount={1}
              customRequest={handleCustomUpload}
              onChange={({ fileList: fl }) => setFileList(fl)}
              onPreview={handlePreview}
              onRemove={() => {
                setCoverImage("");
                setFileList([]);
              }}
              // broaden if needed: "image/*"
              accept="image/jpeg,image/png,image/webp"
            >
              {fileList.length >= 1 ? null : (
                <div>
                  <PlusOutlined />
                  <div style={{ marginTop: 8 }}>Upload</div>
                </div>
              )}
            </Upload>
          </ImgCrop>

          <Modal
            open={previewOpen}
            title="Preview"
            footer={null}
            onCancel={() => setPreviewOpen(false)}
          >
            <img alt="cover" style={{ width: "100%" }} src={previewImage} />
          </Modal>
        </div>

        <div className={internStyles.preQ}>
          <span>Pre Requisites *</span>
          <ChipInput
            initialValue={preRequisites}
            onChipsChange={setPreRequisites}
          />
        </div>

        <div className={internStyles.difficulty}>
          <span>Difficulty</span>
          <div>
            <Radio.Group
              value={difficulty}
              onChange={(e) => setDifficulty(e.target.value)}
            >
              <Radio.Button value="Beginner">Beginner</Radio.Button>
              <Radio.Button value="Intermediate">Intermediate</Radio.Button>
              <Radio.Button value="Advanced">Advanced</Radio.Button>
            </Radio.Group>
          </div>
        </div>

        <div className={internStyles.price}>
          <span>Price</span>
          <div className={internStyles.priceIn}>
            <span>₹</span>
            <input
              placeholder="0.00"
              value={price}
              onChange={(e) => setPrice(e.target.value)}
            />
          </div>
        </div>

        <div className={internStyles.buttonsContainer}>
          <Button type="primary" onClick={handleSave}>
            {params.createInternship?.split("/")?.join("") !== "newCourse"
              ? "Update"
              : "Save"}
          </Button>
          <Button
            type="primary"
            onClick={() =>
              nav.replace(`/course/${params.createInternship}/new`)
            }
          >
            Next
          </Button>
        </div>
      </div>
    </div>
  );
};

export default Page;

        restUrl,

          setImages((prev) => ({ ...prev, [type + "Image"]: s3Url }));

        uid: file.uid || String(Date.now()),
        name: file.name,
        status: "done",
        url,

      if (type === "cover") setFileList([fileObj]);
      else if (type === "thumbnail") setThumbnailFileList([fileObj]);

                setCourseBasic((prev) => ({
                  ...prev,
                  difficulty: e.target.value,

              buttonStyle="solid"
            >

              <Radio.Button value="Intermediate">Intermediate</Radio.Button>
              <Radio.Button value="Advanced">Advanced</Radio.Button>
            </Radio.Group>
          </div>
        </div>

        {/* Learning Outcomes */}
        <div className={internStyles.sectionHeader}>
          Learning Outcomes & Audience
        </div>

        <div className={internStyles.preQ}>
          <p>Prerequisites</p>
          <div style={{ flex: 1 }}>
            <Select
              mode="tags"
              placeholder="Add prerequisites (press enter to add)"
              value={courseArrays.preRequisites}
              onChange={(val) =>
                setCourseArrays((prev) => ({ ...prev, preRequisites: val }))
              }
              style={{ width: "100%" }}
              tokenSeparators={[","]}
            />
          </div>
        </div>

        <div className={internStyles.preQ}>
          <p>Learning Points (What students will learn)</p>
          <div style={{ flex: 1 }}>
            <Select
              mode="tags"
              placeholder="Add learning outcomes"
              value={courseArrays.learningPoints}
              onChange={(val) =>
                setCourseArrays((prev) => ({ ...prev, learningPoints: val }))
              }
              style={{ width: "100%" }}
              tokenSeparators={[","]}
            />
          </div>
        </div>

        <div className={internStyles.preQ}>
          <p>Skills Covered</p>
          <div style={{ flex: 1 }}>
            <Select
              mode="tags"
              placeholder="Add skills"
              value={courseArrays.skills}

                setCourseArrays((prev) => ({ ...prev, tools: val }))

              tokenSeparators={[","]}
            />
          </div>

        <div className={internStyles.preQ}>
          <p>Target Audience</p>
          <div style={{ flex: 1 }}>
            <Select
              mode="tags"
              placeholder="e.g., Aspiring Data Scientists, Software Developers"
              value={courseArrays.targetAudience}
              onChange={(val) =>
                setCourseArrays((prev) => ({ ...prev, targetAudience: val }))
              }
              style={{ width: "100%" }}
              tokenSeparators={[","]}
            />
          </div>
        </div>

        {/* Course Includes */}
        <div className={internStyles.sectionHeader}>Course Includes</div>

        <div className={internStyles.formRow}>
          <div className={internStyles.formField}>
            <label>Articles</label>
            <InputNumber
              min={0}
              value={courseIncludes.articles}
              onChange={(val) =>
                setCourseIncludes((prev) => ({ ...prev, articles: val }))
              }
              style={{ width: "100%" }}
            />
          </div>

          <div className={internStyles.formField}>
            <label>Downloadable Resources</label>
            <InputNumber
              min={0}
              value={courseIncludes.downloadableResources}
              onChange={(val) =>
                setCourseIncludes((prev) => ({
                  ...prev,
                  downloadableResources: val,
                }))
              }
              style={{ width: "100%" }}
            />
          </div>
        </div>

        <div className={internStyles.formRow}>
          <div className={internStyles.formField}>
            <label>Coding Exercises</label>
            <InputNumber
              min={0}
              value={courseIncludes.codingExercises}
              onChange={(val) =>
                setCourseIncludes((prev) => ({ ...prev, codingExercises: val }))
              }
              style={{ width: "100%" }}
            />
          </div>

          <div className={internStyles.formField}>
            <label>Quizzes</label>
            <InputNumber
              min={0}
              value={courseIncludes.quizzes}
              onChange={(val) =>
                setCourseIncludes((prev) => ({ ...prev, quizzes: val }))
              }
              style={{ width: "100%" }}
            />
          </div>
        </div>

        <div className={internStyles.checkboxGroup}>
          <label className={internStyles.checkboxLabel}>
            <input
              type="checkbox"
              checked={courseIncludes.lifetimeAccess}
              onChange={(e) =>
                setCourseIncludes((prev) => ({
                  ...prev,
                  lifetimeAccess: e.target.checked,
                }))
              }
            />
            <span>Lifetime Access</span>
          </label>

          <label className={internStyles.checkboxLabel}>
            <input
              type="checkbox"
              checked={courseIncludes.certificateOfCompletion}
              onChange={(e) =>
                setCourseIncludes((prev) => ({
                  ...prev,
                  certificateOfCompletion: e.target.checked,
                }))
              }
            />
            <span>Certificate of Completion</span>
          </label>

          <label className={internStyles.checkboxLabel}>
            <input
              type="checkbox"
              checked={courseIncludes.jobAssistance}
              onChange={(e) =>
                setCourseIncludes((prev) => ({
                  ...prev,
                  jobAssistance: e.target.checked,
                }))
              }
            />
            <span>Job Assistance</span>
          </label>
        </div>

        {/* Pricing Section */}
        <div className={internStyles.sectionHeader}>Pricing</div>

        <div className={internStyles.formRow}>
          <div className={internStyles.formField}>
            <label>Current Price *</label>
            <div className={internStyles.priceIn}>
              <span>₹</span>
              <input
                placeholder="699"
                value={pricing.currentPrice}
                onChange={(e) =>
                  setPricing((prev) => ({
                    ...prev,
                    currentPrice: e.target.value,
                  }))
                }
              />
            </div>
          </div>

          <div className={internStyles.formField}>
            <label>Original Price</label>
            <div className={internStyles.priceIn}>
              <span>₹</span>
              <input
                placeholder="5999"
                value={pricing.originalPrice}
                onChange={(e) =>
                  setPricing((prev) => ({
                    ...prev,
                    originalPrice: e.target.value,
                  }))
                }
              />
            </div>
          </div>
        </div>

        <div className={internStyles.formRow}>
          <div className={internStyles.formField}>
            <label>Coupon Code</label>
            <input
              placeholder="e.g., DATA_PRO200"
              value={pricing.couponCode}
              onChange={(e) =>
                setPricing((prev) => ({ ...prev, couponCode: e.target.value }))
              }
            />
          </div>

          <div className={internStyles.formField}>
            <label>Coupon Discount (₹)</label>
            <InputNumber
              min={0}
              value={pricing.couponDiscount}
              onChange={(val) =>
                setPricing((prev) => ({ ...prev, couponDiscount: val }))
              }
              style={{ width: "100%" }}
              placeholder="200"
            />
          </div>
        </div>

        {pricing.currentPrice && pricing.couponDiscount > 0 && (
          <div className={internStyles.priceSummary}>
            <Tag color="green">
              Final Price: ₹
              {parseFloat(pricing.currentPrice || 0) -
                parseFloat(pricing.couponDiscount || 0)}
            </Tag>
          </div>
        )}

        {/* Media Section */}
        <div className={internStyles.sectionHeader}>Media Assets</div>

        <div className={internStyles.coverImage_container}>
          <div className={internStyles.sd}>Cover Image (16:9):</div>
          <ImgCrop
            aspect={16 / 9}
            rotate
            zoom
            quality={0.8}
            beforeCrop={validateBeforeCrop}
            cropperProps={{ restrictPosition: false }}
            modalTitle="Edit Cover Image"
          >
            <Upload
              listType="picture-card"
              fileList={fileList}
              maxCount={1}
              customRequest={(options) => handleCustomUpload(options, "cover")}
              onChange={({ fileList: fl }) => setFileList(fl)}
              onPreview={handlePreview}
              onRemove={() => {
                setImages((prev) => ({ ...prev, coverImage: "" }));
                setFileList([]);
              }}
              accept="image/jpeg,image/png,image/webp"
            >
              {fileList.length >= 1 ? null : (
                <div>
                  <PlusOutlined />
                  <div style={{ marginTop: 8 }}>Upload</div>
                </div>
              )}
            </Upload>
          </ImgCrop>
        </div>

        <div className={internStyles.coverImage_container}>
          <div className={internStyles.sd}>Thumbnail (1:1):</div>
          <ImgCrop
            aspect={1}
            rotate
            zoom
            quality={0.8}
            beforeCrop={validateBeforeCrop}
            cropperProps={{ restrictPosition: false }}
            modalTitle="Edit Thumbnail"
          >
            <Upload
              listType="picture-card"
              fileList={thumbnailFileList}
              maxCount={1}
              customRequest={(options) =>
                handleCustomUpload(options, "thumbnail")
              }
              onChange={({ fileList: fl }) => setThumbnailFileList(fl)}
              onPreview={handlePreview}
              onRemove={() => {
                setImages((prev) => ({ ...prev, thumbnailImage: "" }));
                setThumbnailFileList([]);
              }}
              accept="image/jpeg,image/png,image/webp"
            >
              {thumbnailFileList.length >= 1 ? null : (
                <div>
                  <PlusOutlined />
                  <div style={{ marginTop: 8 }}>Upload</div>
                </div>
              )}
            </Upload>
          </ImgCrop>
        </div>

        <div className={internStyles.coverImage_container}>
          <div className={internStyles.sd}>Banner Image (21:9):</div>
          <ImgCrop
            aspect={21 / 9}
            rotate
            zoom
            quality={0.8}
            beforeCrop={validateBeforeCrop}
            cropperProps={{ restrictPosition: false }}
            modalTitle="Edit Banner"
          >
            <Upload
              listType="picture-card"
              fileList={bannerFileList}
              maxCount={1}
              customRequest={(options) => handleCustomUpload(options, "banner")}
              onChange={({ fileList: fl }) => setBannerFileList(fl)}
              onPreview={handlePreview}
              onRemove={() => {
                setImages((prev) => ({ ...prev, bannerImage: "" }));
                setBannerFileList([]);
              }}
              accept="image/jpeg,image/png,image/webp"
            >
              {bannerFileList.length >= 1 ? null : (
                <div>
                  <PlusOutlined />
                  <div style={{ marginTop: 8 }}>Upload</div>
                </div>
              )}
            </Upload>
          </ImgCrop>
        </div>

        <Modal
          open={previewOpen}
          title="Preview"
          footer={null}
          onCancel={() => setPreviewOpen(false)}
        >
          <img alt="preview" style={{ width: "100%" }} src={previewImage} />
        </Modal>

        {/* SEO Section */}
        <div className={internStyles.sectionHeader}>SEO & Metadata</div>

        <div className={internStyles.internTtle}>
          <span>Meta Title</span>
          <input
            placeholder="SEO optimized title"
            value={seo.metaTitle}
            onChange={(e) =>
              setSeo((prev) => ({ ...prev, metaTitle: e.target.value }))
            }
            maxLength={60}
          />
          <small>{seo.metaTitle.length}/60 characters</small>
        </div>

        <div className={internStyles.formField}>
          <label>Meta Description</label>
          <TextArea
            placeholder="SEO meta description"
            value={seo.metaDescription}
            onChange={(e) =>
              setSeo((prev) => ({ ...prev, metaDescription: e.target.value }))
            }
            maxLength={160}
            rows={3}
            showCount
          />
        </div>

        <div className={internStyles.preQ}>
          <span>SEO Keywords</span>
          <div style={{ flex: 1 }}>
            <Select
              mode="tags"
              placeholder="Add SEO keywords"
              value={courseArrays.keywords}
              onChange={(val) =>
                setCourseArrays((prev) => ({ ...prev, keywords: val }))
              }
              style={{ width: "100%" }}
              tokenSeparators={[","]}
              maxTagCount="responsive"
              maxTagTextLength={25}
            />
          </div>
        </div>

        {/* Status & Visibility */}
        <div className={internStyles.sectionHeader}>Status & Visibility</div>

        <div className={internStyles.formRow}>
          <div className={internStyles.formField}>
            <label>Course Status</label>
            <Select
              value={visibility.status}
              onChange={(val) =>
                setVisibility((prev) => ({ ...prev, status: val }))
              }
              style={{ width: "100%" }}
            >
              <Option value="active">Active</Option>
              <Option value="draft">Draft</Option>
              <Option value="archived">Archived</Option>
            </Select>
          </div>

          <div className={internStyles.formField}>
            <div className={internStyles.checkboxGroup}>
              <label className={internStyles.checkboxLabel}>
                <input
                  type="checkbox"
                  checked={visibility.featured}
                  onChange={(e) =>
                    setVisibility((prev) => ({
                      ...prev,
                      featured: e.target.checked,
                    }))
                  }
                />
                <span>Featured Course</span>
              </label>

              <label className={internStyles.checkboxLabel}>
                <input
                  type="checkbox"
                  checked={visibility.trending}
                  onChange={(e) =>
                    setVisibility((prev) => ({
                      ...prev,
                      trending: e.target.checked,
                    }))
                  }
                />
                <span>Trending Course</span>
              </label>
            </div>
          </div>
        </div>

        {/* Action Buttons */}
        <div className={internStyles.buttonsContainer}>
          <Button type="default" onClick={() => nav.back()}>
            Cancel
          </Button>
          <Button type="primary" onClick={handleSave}>
            {params.createInternship?.split("/")?.join("") !== "newCourse"
              ? "Update Course"
              : "Save & Continue"}
          </Button>
          <Button
            type="primary"
            onClick={() =>
              nav.replace(`/course/${params.createInternship}/new`)
            }
          >
            Next: Add Curriculum
          </Button>
        </div>
      </div>
    </div>
  );
};

export default Page;
